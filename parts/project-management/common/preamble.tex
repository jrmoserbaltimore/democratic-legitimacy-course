\newganttlinktype{st-st}{
    \ganttsetstartanchor{on left=0.5}
    \ganttsetendanchor{on left=0.5}
    \draw [->,thick] (\xLeft, \yUpper)
    -| (\xLeft - 0.5, \yLower)
    -- (\xRight, \yLower);
}

\newganttlinktype{fn-fn}{
    \ganttsetstartanchor{on right=0.75}
    \ganttsetendanchor{on right=0.5}

    -| (\xRight + 0.5, \yLower)
    -- (\xRight, \yLower);
}
