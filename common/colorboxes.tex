
% Skins: for frame styles and attaching box to top etc.
\tcbuselibrary{skins}

%\pgfplotsset{compat=1.17}

\tcbset{
    corner title/main style/.style={},
    corner title/title style/.style={},
    corner title/main/.style={/tcb/corner title/main style/.style={#1}},
    corner title/title/.style={/tcb/corner title/title style/.style={#1}},
    corner title/append main/.style={/tcb/corner title/main style/.append style={#1}},
    corner title/append title/.style={/tcb/corner title/title style/.append style={#1}},
    corner title/.style={
        corner title/.cd, #1,
        /tcb/.cd,
        enhanced,
        corner title/main style,
        attach boxed title to top left={xshift=0.5cm, yshift=-12pt},
        boxed title style={
            outer arc=10pt,
            arc=10pt,
            top=3pt,
            bottom=3pt,
            corner title/title style,
        },
        top=12pt,
    }
}


% A frame for example elections

\newcounter{election}
\newrefformat{ele}{Election \ref{#1}}
\newtcolorbox{election}[1][]{
    corner title={
        main={
            colframe = blue!70!black,
            colback = yellow!40,
        },
        title = {
            colback=black,
            coltext=white
        },
    },
    before title={\global\refstepcounter{election}},
    before upper={\addtocounter{election}{-1}\refstepcounter{election}},
    title={Election \arabic{election}},
    fonttitle=\sffamily\bfseries,
    #1,
}
